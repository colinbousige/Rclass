  • Write a function returning the derivative \(\partial y/\partial x\) given two vectors x and y
a <- c(runif(10,1,10))  
b <- c(runif(10,2,20))
i <- 1:length(a)

der <- function(x,y) {
  tibble(x=x, y=y, Der_x = x[i+1]-x[i],Der_y = y[i+1]-y[i],Dyx= Der_y/Der_x)
}
  der(a,b)
